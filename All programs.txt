16.)Write a program to demonstrate the difference between pre-increment and post-increment
#include <stdio.h>
int main()
{
    int a = 5, b = 5;

    printf("Initial values:\n");
    printf("a = %d, b = %d\n", a, b);

    // Pre-increment
    printf("\nUsing pre-increment (++a):\n");
    printf("Value returned: %d\n", ++a); 
    printf("Final value of a: %d\n", a);

    // Post-increment
    printf("\nUsing post-increment (b++):\n");
    printf("Value returned: %d\n", b++); 
    printf("Final value of b: %d\n", b);

    return 0;
}
